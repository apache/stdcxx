Summary

Locale convenience function that converts a character to lower case

Synopsis

 
#include <locale>

namespace std {
  template <class charT> 
  charT tolower(charT c, const locale& loc);
}

Description

The tolower() function returns the parameter c after converting it to lower case. The conversion is made using the ctype facet from the locale parameter.

Example

 
    //
//  toupper.cpp
//
 
#include <iomanip>    // for setw
#include <iostream>   // for cout, endl

int main ()
{
  std::cout << std::oct;
  std::cout.fill ('0');

  // compute tolower and toupper of printable ASCII characters
  for (int c = ' '; c != '~' + 1; ++c)
    std::cout << "std::toupper/lower ('\\" << std::setw (3) 
              << c
              << "', std::locale ()) = '"
              << std::toupper (char (c), std::cout.getloc ())
              << "' / "
              << std::tolower (char (c), std::cout.getloc ())
              << "'\n";
 
    return 0;
}
